Property: customEventPrefix (default is null)
This property is used to override the widgetEventPrefix provided by the Widget Factory. widgetEventPrefix is basically the name of the widget (paginateddropdown in this case) which would be prefixed to any event that is exposed by the widget (itemselected in this case). So, if we do not override customEventPrefix then you can bind the 'paginateddropdownitemselected' event to the element which would get fired when ever we click an item in the drop down.
Demo: default behaviour
Sample Code:
$( "#x-demo-1" ).paginateddropdown({
	url: "assets/cricketers.json",
	valueProperty: "name",
	itemTemplate: "<div title='{@:email}'><div>{@:name}</div></div>",
	filterJSON: {startsWith:true}
}).bind( "paginateddropdownitemselected", function( event, data ) {
	if(window.console) {
		console.log("callback - paginateddropdownitemselected");
		console.log(event);
		console.log(data);
	}
});
Demo: Setting customEventPrefix to "pdd"
Sample Code:
$( "#x-demo-2" ).paginateddropdown({
	url: "assets/cricketers.json",
	valueProperty: "name",
	itemTemplate: "<div title='{@:email}'><div>{@:name}</div></div>",
	customEventPrefix: "pdd",
	filterJSON: {startsWith:true}
}).bind( "pdditemselected", function( event, data ) {
	if(window.console) {
		console.log("callback - pdditemselected");
		console.log(event);
		console.log(data);
	}
});
